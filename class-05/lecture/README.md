 *      h1

 *      nav
 *        li
 *          a
 *    article
 *      section .intro
 *        h1
 *        p
 *      section .story
 *        h1
 *        h2
 *        p
 *    footer
 *      nav
 *        li
 *          a
 */  

/*
 * font-family: 'Niconne', cursive;
 *
 * font-family: 'Vast Shadow', cursive;
 *
 * font-family: 'Sonsie One', cursive;
 *
 * font-family: 'Chango', cursive;
 *
 * font-family: 'Open Sans Condensed', sans-serif;
 */

}

}

}

}
